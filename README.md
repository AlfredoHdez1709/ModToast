### Add it in your root build.gradle at the end of repositories:

    allprojects {
        repositories {
          ...
          maven { url 'https://jitpack.io' }
        }
      }
### Add the dependency

      dependencies {
                implementation 'com.github.AlfredoHdez1709:ModToast:0.1'
        }
